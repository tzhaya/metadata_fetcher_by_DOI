(ISSN as text, apikey as text) =>
    let
    // Open Policy Finder APIからISSNを送信してJSONを取得
        queryparameter = [ #"item-type" = "publication", format = "Json" , identifier = ISSN ],
        OpfSource = Web.Contents("https://api.openpolicyfinder.jisc.ac.uk" ,
        [   RelativePath="retrieve_by_id",
            Headers = [#"x-api-key" = apikey],
            Query =  queryparameter,
            ManualStatusHandling={400, 404, 500}
            ]
        ),
        status = Value.Metadata(OpfSource)[Response.Status],
    // [items]{0}[system_metadata][uri]、ポリシーがあるURLを取得
        oauri = Json.Document(OpfSource)[items]{0}[system_metadata][uri]
    in
        if status = 404 then null else
        oauri
